#include <bits/stdc++.h>
using namespace std;

string all_letrs = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";

map<char, char> buildEncryptMap(int key) {
    map<char, char> dict;
    for (int i = 0; i < all_letrs.length(); i++)
        dict[all_letrs[i]] = all_letrs[(i + key) % all_letrs.length()];
    return dict;
}

map<char, char> buildDecryptMap(int key) {
    map<char, char> dict;
    for (int i = 0; i < all_letrs.length(); i++)
        dict[all_letrs[i]] = all_letrs[(i - key + all_letrs.length()) % all_letrs.length()];
    return dict;
}

string applyMap(string text, map<char, char> dict) {
    string result;
    for (char& c : text) {
        if (dict.find(c) != dict.end())
            result += dict[c];
        else
            result += c;
    }
    return result;
}

int main() {
    int key = 4;
    string plain_txt = "Hello World I am studying NS";

    string cipher_txt  = applyMap(plain_txt,  buildEncryptMap(key));
    string decrypt_txt = applyMap(cipher_txt, buildDecryptMap(key));

    cout << "Original Text    : " << plain_txt   << endl;
    cout << "Cipher Text      : " << cipher_txt  << endl;
    cout << "Recovered Text   : " << decrypt_txt << endl;

    return 0;
}
